\textbf{Resumen}

\begin{equation*}
  V_o(V_i)=
  \begin{cases}
    \frac{V_i + 6}{4} \quad
      \phantom{\frac{V_i + 10}{8}\frac{V_i - 2}{4}}
      \text{si} \quad V_i \leq -2
      \phantom{\text{si} \quad -2 < V_i \leq 14\text{si} \quad V_i > 14}
      (\ref{p3:offoff})
    \\
    \frac{V_i + 10}{8} \quad
      \phantom{\frac{V_i + 6}{4}\frac{V_i - 2}{4}}
      \text{si} \quad -2 < V_i \leq 14
      \phantom{\text{si} \quad V_i \leq -2\text{si} \quad V_i > 14}
      (\ref{p3:offon})
    \\
    \frac{V_i - 2}{4} \quad
      \phantom{\frac{V_i + 6}{4}\frac{V_i + 10}{8}}
      \text{si} \quad V_i > 14
      \phantom{\text{si} \quad V_i \leq -2\text{si} \quad -2 < V_i \leq 14}
      (\ref{p3:onon})
  \end{cases}
\end{equation*}
